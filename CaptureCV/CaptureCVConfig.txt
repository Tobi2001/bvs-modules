# This is an example config file for CaptureCV. It will be linked to
# [build/]bin/CaptureCVConfig.txt so you can get started right away.
#
# For more information on syntax as well as further capabilities, please see
# 'libbvs/include/bvs/config.h' or 'DOXYGEN:Classes->BVS::Config'.

[capture]

# number of (capture/video/file inputs/outputs)
numNodes = 2

# selects the mode to use, possible options are (actually only the first character is important):
# CAMERA      -- use CAMERAs as input sources
# VIDEO       -- use VIDEOs given in 'videoFiles'
# IMAGE       -- use IMAGEs, names are generated by 'imageFiles'
# RECORD_VIDEO -- record to VIDEOs given in 'videoFiles'
# SAVE_IMAGE -- save to IMAGEs, names are generated by 'imageFiles'
#
mode = CAMERA

# list of files to use for recording or capturing from
videoFiles = video_1.avi, video_2.avi

# image files naming scheme
# use '{FRAME}' as a placeholder for the frame number, and '{NODE}' as a
# placeholder for the node id, also can contain path elements
# extension must be supported by opencv
imageFiles = images/frame_{FRAME}_{NODE}.jpg

# camera mode, set to -1 to disable explicitely setting it
# for grasshopper: 0-1: color, 2: black and white, 3: weiiiiiird...
cameraMode = 2

# camera fps (must be supported by camera), set to -1 to disable
cameraFPS = -1

# video recording codec to use, must be supported by system (opencv uses ffmpeg)
videoFOURCC = "MJPG"

# video fps to set in video metadata, MUST BE SET
videoFPS = 0.0

# video width to set in video metadata, 0 for auto
videoWidth = 0

# video height to set in video metadata, 0 for auto
videoHeight = 0

# record with color or black and white (must match your input)
videoColor = false



# alternative configurations (use by creating modules with these section ids or create your own)
[video]
numNodes = 2
mode = VIDEO
videoFiles = video_1.avi, video_2.avi

[video_writer]
numNodes = 2
mode = RECORD_VIDEO
videoFiles = video_1.avi, video_2.avi
videoFOURCC = "MJPG"
videoFPS = 15
videoWidth = 0
videoHeight = 0
videoColor = false

[image]
numNodes = 2
mode = IMAGE
imageFiles = images/frame_{FRAME}_{NODE}.jpg

[image_writer]
numNodes = 2
mode = SAVE_IMAGE
imageFiles = images/frame_{FRAME}_{NODE}.jpg
